800DA840 overlay table

struct OverlayEntry
{
    void *location;
    u16   objectNum;
    u16   _pad;
    void *fakeMemStart;
    void *fakeMemEnd;
};

/////////////////

linker routine: func_80081430(void *overlay_block, void *meta_section);

Overlay file structure:

    u8  text[textSize]
    u8  data[dataSize]
    u8  rodata[rodataSize]
    meta {
        u32 textSize
        u32 dataSize
        u32 rodataSize
        u32 bssSize
        u32 numRelocations
        u32 relocations[numRelocations]
        u32 metaSize
    }

    The position of the meta section is calculated by
    subtracting metaSize from the total size of the overlay file.
    
    .bss is allocated immediately after the overlay file in memory.

    Relocation bit fields:

        ss -- rrrr oooooooooooooooooooooooo

        s: Section (0 = invalid?, 1 = .text, 2 = .data, 3 = .rodata)
        -: Unused bits?
        r: Relocation type (4 = R_MIPS_26, 5 = R_MIPS_HI16, 6 = R_MIPS_LO16)
        o: Offset
